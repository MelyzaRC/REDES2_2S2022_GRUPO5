!*--------------------------------------------------
!INTERFACES TIENEN QUE ESTAR EN MODO ACCESO PARA CONFIGURAR PORT SECURITY
!INTERFACES DE VENTAS
enable
configure terminal
interface f 0/4
!Port-security viene desactivado por defecto asi que se debe activar 
switchport port-security
switchport port-security mac-address sticky
switchport port-security maximum 5
end


!INTERFACES DE DISTRIBUCION PARA TODOS LOS QUE SE CONECTEN A HOSTS
enable
configure terminal
interface f 0/7
switchport port-security
!Sticky es para que aprenda la MAC que esta configurada al puerto y la toma como la unica permitida 
switchport port-security mac-address sticky
switchport port-security maximum 1
switchport port-security violation shutdown
end

!INTERFACES DE ADMINISTRACION
enable
configure terminal
interface f 0/6
switchport port-security
switchport port-security mac-address
switchport port-security maximum 1
switchport port-security violation shutdown
end



!PARA VER LA CONFIGURACION DE PORT SECURITY
show port-security interface f #/#
